-- ═══════════════════════════════════════════════════════════════════ -- 🌳 해누리 질문 과수원 — 전체 마이그레이션 스크립트 -- 📅 2026.04 (v4 — 숲 확장 포함) -- ═══════════════════════════════════════════════════════════════════ -- -- 이 스크립트는 Supabase SQL Editor에서 한 번에 실행하면 됩니다. -- 모든 문이 `IF NOT EXISTS` / `CREATE OR REPLACE` 이므로 -- 여러 번 실행해도 안전합니다. -- -- 실행 후 맨 아래의 확인 쿼리로 데이터가 잘 있는지 볼 수 있어요. -- -- ⚠️ 백업 권장: Supabase Dashboard → Database → Backups -- ═══════════════════════════════════════════════════════════════════ -- ─────────────────────────────────────────────────────────────────── -- 1. profiles 테이블 확장 -- ─────────────────────────────────────────────────────────────────── ALTER TABLE profiles ADD COLUMN IF NOT EXISTS seeds int DEFAULT 0, ADD COLUMN IF NOT EXISTS unlocked_items text[] DEFAULT ARRAY['default','none']::text[], ADD COLUMN IF NOT EXISTS active_tree text DEFAULT 'default', ADD COLUMN IF NOT EXISTS active_companion text, ADD COLUMN IF NOT EXISTS streak_days int DEFAULT 0, ADD COLUMN IF NOT EXISTS last_active date, ADD COLUMN IF NOT EXISTS waters_given int DEFAULT 0, ADD COLUMN IF NOT EXISTS active_sub text; -- ─────────────────────────────────────────────────────────────────── -- 2. questions 테이블 확장 (🆕 tree_id, fruit_x, fruit_y 포함) -- ─────────────────────────────────────────────────────────────────── ALTER TABLE questions ADD COLUMN IF NOT EXISTS tags text[] DEFAULT ARRAY[]::text[], ADD COLUMN IF NOT EXISTS fruit_emoji text, ADD COLUMN IF NOT EXISTS is_anonymous boolean DEFAULT false, ADD COLUMN IF NOT EXISTS water_count int DEFAULT 0, ADD COLUMN IF NOT EXISTS watered_by uuid[] DEFAULT ARRAY[]::uuid[], ADD COLUMN IF NOT EXISTS branch_count int DEFAULT 0, ADD COLUMN IF NOT EXISTS fruit_x numeric, -- 열매 커스텀 위치 X (드래그 이동) ADD COLUMN IF NOT EXISTS fruit_y numeric, -- 열매 커스텀 위치 Y ADD COLUMN IF NOT EXISTS tree_id bigint, -- 🌲 소속 나무 (trees.id) ADD COLUMN IF NOT EXISTS starred boolean DEFAULT false, -- ⭐ 교사 뽑은 질문 (명예의 전당) ADD COLUMN IF NOT EXISTS hint text; -- ─────────────────────────────────────────────────────────────────── -- 3. 🌲 trees 테이블 (숲 확장 — 과목당 여러 나무) -- ─────────────────────────────────────────────────────────────────── CREATE TABLE IF NOT EXISTS trees ( id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY, user_id uuid REFERENCES profiles(id) ON DELETE CASCADE, subject_id text NOT NULL, name text NOT NULL, icon text DEFAULT '🌳', sort_order int DEFAULT 1, created_at timestamptz DEFAULT now() ); -- questions.tree_id → trees.id 외래키 추가 (이미 있으면 skip) DO $$ BEGIN IF NOT EXISTS ( SELECT 1 FROM information_schema.table_constraints WHERE constraint_name = 'questions_tree_id_fkey' AND table_name = 'questions' ) THEN ALTER TABLE questions ADD CONSTRAINT questions_tree_id_fkey FOREIGN KEY (tree_id) REFERENCES trees(id) ON DELETE CASCADE; END IF; END$$; -- ─────────────────────────────────────────────────────────────────── -- 4. branches (꼬리질문) 테이블 — 없으면 생성 -- ─────────────────────────────────────────────────────────────────── CREATE TABLE IF NOT EXISTS branches ( id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY, question_id bigint REFERENCES questions(id) ON DELETE CASCADE, author_id uuid REFERENCES profiles(id) ON DELETE CASCADE, author_name text, text text NOT NULL, answer text, created_at timestamptz DEFAULT now() ); -- ─────────────────────────────────────────────────────────────────── -- 5. tree_state (스티커 저장용) -- ─────────────────────────────────────────────────────────────────── CREATE TABLE IF NOT EXISTS tree_state ( user_id uuid PRIMARY KEY REFERENCES profiles(id) ON DELETE CASCADE, stickers jsonb DEFAULT '[]'::jsonb, updated_at timestamptz DEFAULT now() ); -- ─────────────────────────────────────────────────────────────────── -- 6. 🧑‍🏫 teachers 테이블 (교사 계정 — 과목 매핑) -- ─────────────────────────────────────────────────────────────────── CREATE TABLE IF NOT EXISTS teachers ( id bigint GENERATED ALWAYS AS IDENTITY PRIMARY KEY, auth_id uuid UNIQUE NOT NULL, -- Supabase Auth user ID name text NOT NULL, -- 선생님 성함 subject_id text NOT NULL, -- 담당 과목 ID (subjects.id와 매칭) subject_label text, -- 과목 이름 (표시용) subject_icon text DEFAULT '📚', -- 과목 아이콘 grade int, -- 담당 학년 (NULL이면 전학년) created_at timestamptz DEFAULT now() ); -- questions 테이블에 starred 컬럼 추가 (이미 있으면 skip) ALTER TABLE questions ADD COLUMN IF NOT EXISTS starred boolean DEFAULT false; -- ═══════════════════════════════════════════════════════════════════ -- 🔐 RLS 정책 (Row-Level Security) -- ⚠️ profiles 정책 내에서 profiles를 서브쿼리로 조회하면 무한재귀 발생 -- → SECURITY DEFINER 함수로 우회합니다 -- ═══════════════════════════════════════════════════════════════════ -- 헬퍼 함수: 현재 로그인한 사용자의 학년·반 조회 (RLS 우회) CREATE OR REPLACE FUNCTION get_my_grade_class() RETURNS TABLE(grade int, class_no int) LANGUAGE sql SECURITY DEFINER STABLE AS $$ SELECT grade, class_no FROM profiles WHERE id = auth.uid() LIMIT 1; $$; -- ── profiles: 본인 전체 + 같은 반 읽기 ALTER TABLE profiles ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "profiles_self_all" ON profiles; CREATE POLICY "profiles_self_all" ON profiles FOR ALL USING (auth.uid() = id) WITH CHECK (auth.uid() = id); DROP POLICY IF EXISTS "profiles_classmates_read" ON profiles; CREATE POLICY "profiles_classmates_read" ON profiles FOR SELECT USING ( (grade, class_no) IN (SELECT grade, class_no FROM get_my_grade_class()) ); -- ── questions: 본인 전체 + 같은 반 읽기 + 물주기 업데이트 ALTER TABLE questions ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "questions_self_all" ON questions; CREATE POLICY "questions_self_all" ON questions FOR ALL USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id); DROP POLICY IF EXISTS "questions_classmates_read" ON questions; CREATE POLICY "questions_classmates_read" ON questions FOR SELECT USING ( EXISTS ( SELECT 1 FROM profiles owner WHERE owner.id = questions.user_id AND (owner.grade, owner.class_no) IN (SELECT grade, class_no FROM get_my_grade_class()) ) ); DROP POLICY IF EXISTS "questions_classmates_water" ON questions; CREATE POLICY "questions_classmates_water" ON questions FOR UPDATE USING ( EXISTS ( SELECT 1 FROM profiles owner WHERE owner.id = questions.user_id AND (owner.grade, owner.class_no) IN (SELECT grade, class_no FROM get_my_grade_class()) ) ) WITH CHECK ( EXISTS ( SELECT 1 FROM profiles owner WHERE owner.id = questions.user_id AND (owner.grade, owner.class_no) IN (SELECT grade, class_no FROM get_my_grade_class()) ) ); -- ── 🌲 trees: 본인만 + 같은 반 읽기 ALTER TABLE trees ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "trees_self_all" ON trees; CREATE POLICY "trees_self_all" ON trees FOR ALL USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id); DROP POLICY IF EXISTS "trees_classmates_read" ON trees; CREATE POLICY "trees_classmates_read" ON trees FOR SELECT USING ( EXISTS ( SELECT 1 FROM profiles owner WHERE owner.id = trees.user_id AND (owner.grade, owner.class_no) IN (SELECT grade, class_no FROM get_my_grade_class()) ) ); -- ── branches: 본인 작성 전체 + 같은 반 읽기 ALTER TABLE branches ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "branches_self_write" ON branches; CREATE POLICY "branches_self_write" ON branches FOR ALL USING (auth.uid() = author_id) WITH CHECK (auth.uid() = author_id); DROP POLICY IF EXISTS "branches_classmates_read" ON branches; CREATE POLICY "branches_classmates_read" ON branches FOR SELECT USING ( EXISTS ( SELECT 1 FROM questions q JOIN profiles owner ON owner.id = q.user_id WHERE q.id = branches.question_id AND (owner.grade, owner.class_no) IN (SELECT grade, class_no FROM get_my_grade_class()) ) ); -- ── subjects: 공용은 모두 읽기, 개인 것은 본인만 ALTER TABLE subjects ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "subjects_public_read" ON subjects; CREATE POLICY "subjects_public_read" ON subjects FOR SELECT USING (user_id IS NULL OR user_id = auth.uid()); DROP POLICY IF EXISTS "subjects_self_write" ON subjects; CREATE POLICY "subjects_self_write" ON subjects FOR ALL USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id); -- ── tree_state: 본인만 ALTER TABLE tree_state ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "tree_state_self" ON tree_state; CREATE POLICY "tree_state_self" ON tree_state FOR ALL USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id); -- ── teachers: 본인 프로필 전체 + 다른 교사 읽기 ALTER TABLE teachers ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "teachers_self_all" ON teachers; CREATE POLICY "teachers_self_all" ON teachers FOR ALL USING (auth.uid() = auth_id) WITH CHECK (auth.uid() = auth_id); DROP POLICY IF EXISTS "teachers_read_all" ON teachers; CREATE POLICY "teachers_read_all" ON teachers FOR SELECT USING (true); -- 모든 인증 사용자가 교사 목록 조회 가능 -- ═══════════════════════════════════════════════════════════════════ -- ⚡ 인덱스 (성능 개선) -- ═══════════════════════════════════════════════════════════════════ CREATE INDEX IF NOT EXISTS idx_questions_user_subject ON questions(user_id, subject_id); CREATE INDEX IF NOT EXISTS idx_questions_tree ON questions(tree_id); CREATE INDEX IF NOT EXISTS idx_questions_starred ON questions(starred) WHERE starred = true; CREATE INDEX IF NOT EXISTS idx_questions_created_at ON questions(created_at DESC); CREATE INDEX IF NOT EXISTS idx_questions_subject ON questions(subject_id); CREATE INDEX IF NOT EXISTS idx_branches_question ON branches(question_id); CREATE INDEX IF NOT EXISTS idx_branches_author ON branches(author_id); CREATE INDEX IF NOT EXISTS idx_trees_user_subject ON trees(user_id, subject_id); CREATE INDEX IF NOT EXISTS idx_profiles_grade_class ON profiles(grade, class_no); CREATE INDEX IF NOT EXISTS idx_teachers_auth ON teachers(auth_id); -- ═══════════════════════════════════════════════════════════════════ -- 🧑‍🏫 교사 계정이 학생 데이터를 조회할 수 있도록 (인증된 사용자 READ) -- ═══════════════════════════════════════════════════════════════════ -- 교사는 학생과 다른 Auth 계정이므로 기존 "같은 반" 정책에 해당하지 않음 -- 아래 정책으로 인증된 사용자(교사 포함)가 전체 데이터를 SELECT 가능 DROP POLICY IF EXISTS "authenticated_read_profiles" ON profiles; CREATE POLICY "authenticated_read_profiles" ON profiles FOR SELECT USING (auth.role() = 'authenticated'); DROP POLICY IF EXISTS "authenticated_read_questions" ON questions; CREATE POLICY "authenticated_read_questions" ON questions FOR SELECT USING (auth.role() = 'authenticated'); DROP POLICY IF EXISTS "authenticated_read_branches" ON branches; CREATE POLICY "authenticated_read_branches" ON branches FOR SELECT USING (auth.role() = 'authenticated'); DROP POLICY IF EXISTS "authenticated_read_subjects" ON subjects; CREATE POLICY "authenticated_read_subjects" ON subjects FOR SELECT USING (auth.role() = 'authenticated'); DROP POLICY IF EXISTS "authenticated_read_trees" ON trees; CREATE POLICY "authenticated_read_trees" ON trees FOR SELECT USING (auth.role() = 'authenticated'); -- 교사가 ⭐ 별 토글(starred)을 할 수 있도록 questions UPDATE 허용 DROP POLICY IF EXISTS "authenticated_update_questions" ON questions; CREATE POLICY "authenticated_update_questions" ON questions FOR UPDATE USING (auth.role() = 'authenticated');
